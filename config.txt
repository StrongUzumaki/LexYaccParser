server {

	listen 127.0.0.0:8080    
	location /images/ {
        # Set the root directory to search for the file
        root /data/www;


        # Make an internal redirect if the file is not found
        error_page 404 = /fetch$uri;
    }

    location /fetch/ {
		limit_except	GET POST DELETE
		autoindex	on;
    }
}
